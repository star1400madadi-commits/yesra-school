let students = JSON.parse(localStorage.getItem("students")) || {};
let nextId = parseInt(localStorage.getItem("nextId")) || 1001;

// 🔍 نمایش نتیجه
function login() {
  const id = document.getElementById("id").value;
  const pass = document.getElementById("pass").value;
  const res = document.getElementById("result");

  if (students[id] && students[id].pass === pass) {
    res.innerHTML = "👤 " + students[id].name + "<br>📊 نمره: " + students[id].score;
  } else {
    res.innerHTML = "❌ اطلاعات اشتباه";
  }
}

// 🔐 ورود ادمین
function adminLogin() {
  const pass = document.getElementById("adminPass").value;

  if (pass === "admin123") {
    document.getElementById("adminPanel").style.display = "block";
  } else {
    alert("رمز اشتباه!");
  }
}

// ➕ افزودن دانش‌آموز
function addStudent() {
  const name = document.getElementById("name").value;
  const score = document.getElementById("score").value;
  const pass = document.getElementById("newPass").value;

  if (!name || !score || !pass) {
    alert("همه فیلدها را پر کن");
    return;
  }

  const id = nextId++;
  students[id] = {name: name, score: score, pass: pass};

  localStorage.setItem("students", JSON.stringify(students));
  localStorage.setItem("nextId", nextId);

  document.getElementById("msg").innerHTML = "✅ ثبت شد<br>آیدی: " + id;
}
